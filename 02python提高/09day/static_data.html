from socket import socket
from socket import AF_INET
from socket import SOCK_STREAM
from socket import SOL_SOCKET
from socket import SO_REUSEADDR


def request_handler(sock):
    """处理客户端请求"""

    request_data = sock.recv(4096)
    if not request_data:
        print('客户端已经断开连接.')
        sock.close()
        return

    response_line = 'HTTP/1.1 200 OK\r\n'
    response_header = 'Server: PythonWebServer1.0\r\n'
    response_content = 'Hello World!'
    response_data = response_line + response_header + '\r\n' + response_content

    # 发送响应报文
    sock.send(response_data.encode())
    sock.close()


def main():

    # 创建套接字
    sock = socket(AF_INET, SOCK_STREAM)
    # 设置套接字复用地址
    sock.setsockopt(SOL_SOCKET, SO_REUSEADDR, 1)
    # 绑定IP地址和端口
    sock.bind(('', 10000))
    # 设置被动套接字
    sock.listen(128)

    while True:
        # 等待被连接
        client_sock, client_addr = sock.accept()
        # 处理请求
        request_handler(client_sock)


if __name__ == '__main__':
    main()
